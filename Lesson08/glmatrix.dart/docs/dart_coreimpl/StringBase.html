Code String join(List strings, String separator) #

static String join(List<String> strings, String separator) {
  if (strings.length == 0) return '';
  String s = strings[0];
  for (int i = 1; i < strings.length; i++) {
    s = s + separator + strings[i];
  }
  return s;
}
